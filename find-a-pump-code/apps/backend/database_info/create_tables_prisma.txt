This is made from my create tables, translated over to prisma/sqlite

// schema.prisma
datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

generator client {
  provider = "prisma-client-js"
}

model User {
  id         String       @id @default(uuid()) @map("user_id")
  username   String?      @db.VarChar(40)
  UserConfig UserConfig[]

  @@map("User")
}

model FuelType {
  id        String      @id @default(uuid()) @map("fuel_type_id")
  name      String?     @db.VarChar(30)
  FuelPrice FuelPrice[]
  UserConfig UserConfig[]

  @@map("FuelType")
}

model UserConfig {
  id                String    @id @default(uuid()) @map("config_id")
  userId            String    @map("user_id")
  fuelPreferenceId  String?   @map("fuel_preference_id")
  
  user              User      @relation(fields: [userId], references: [id])
  fuelPreference    FuelType? @relation(fields: [fuelPreferenceId], references: [id])

  @@map("UserConfig")
}

model StationBrand {
  id        String    @id @default(uuid()) @map("station_brand_id")
  brandName String?   @map("brand_name") @db.VarChar(50)
  logoUrl   String?   @map("logo_url") @db.VarChar(150)
  Station   Station[]

  @@map("StationBrand")
}

model Location {
  id            String    @id @default(uuid()) @map("location_id")
  googleMapsUrl String?   @map("google_maps_url") @db.VarChar(150)
  street        String?   @db.VarChar(30)
  city          String?   @db.VarChar(30)
  zip           String?   @db.VarChar(30)
  country       String?   @db.VarChar(30)
  lat           Float?
  long          Float?
  Station       Station[]

  @@map("Location")
}

model Station {
  id             String        @id @default(uuid()) @map("station_id")
  locationId     String        @map("location_id")
  stationBrandId String        @map("station_brand_id")
  
  location       Location      @relation(fields: [locationId], references: [id])
  stationBrand   StationBrand  @relation(fields: [stationBrandId], references: [id])
  FuelPrice      FuelPrice[]

  @@map("Station")
}

model FuelPrice {
  id         String   @id @default(uuid()) @map("fuel_price_id")
  stationId  String   @map("station_id")
  fuelTypeId String   @map("fuel_type_id")
  fuelUnit   String?  @map("fuel_unit") @db.VarChar(20)
  fuelPrice  Float?   @map("fuel_price")
  createdAt  DateTime @map("created_at")
  
  station    Station  @relation(fields: [stationId], references: [id])
  fuelType   FuelType @relation(fields: [fuelTypeId], references: [id])

  @@map("FuelPrice")
}
